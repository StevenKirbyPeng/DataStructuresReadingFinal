#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
using namespace std;

// ===================== Binary Tree Traversal =====================
struct TNode {
    char data;
    TNode *left, *right;
    TNode(char d) : data(d), left(nullptr), right(nullptr) {}
};

void preorder(TNode* r) {        // VLR：先拜訪自己，再左，再右
    if (!r) return;
    cout << r->data << ' ';
    preorder(r->left);
    preorder(r->right);
}
void inorder(TNode* r) {         // LVR：左、自己、右
    if (!r) return;
    inorder(r->left);
    cout << r->data << ' ';
    inorder(r->right);
}
void postorder(TNode* r) {       // LRV：左、右、自己
    if (!r) return;
    postorder(r->left);
    postorder(r->right);
    cout << r->data << ' ';
}

// ===================== Max Heap =====================
class MaxHeap {
private:
    vector<int> heap; // heap[0] 不使用，root 放 heap[1]
public:
    MaxHeap() { heap.push_back(0); }
    bool empty() const { return heap.size() == 1; }
    int top() const { return heap[1]; }

    void push(int x) {
        heap.push_back(x);              // Step 1：先放在 complete tree 的最後
        int i = heap.size() - 1;
        while (i > 1 && heap[i/2] < x) { // Step 2：若比 parent 大就往上
            heap[i] = heap[i/2];
            i /= 2;
        }
        heap[i] = x;
    }

    void pop() {
        if (empty()) return;
        int last = heap.back();
        heap.pop_back();
        if (empty()) return;
        int i = 1, child = 2;
        while (child < (int)heap.size()) {
            if (child + 1 < (int)heap.size() && heap[child] < heap[child+1]) child++;
            if (last >= heap[child]) break;
            heap[i] = heap[child];
            i = child;
            child *= 2;
        }
        heap[i] = last;
    }

    void print() const {
        for (int i = 1; i < (int)heap.size(); ++i) cout << heap[i] << ' ';
        cout << '\n';
    }
};

// ===================== BST =====================
struct BNode {
    int key;
    BNode *left, *right;
    BNode(int k) : key(k), left(nullptr), right(nullptr) {}
};

BNode* insertBST(BNode* r, int key) {
    if (!r) return new BNode(key);
    if (key < r->key) r->left = insertBST(r->left, key);
    else if (key > r->key) r->right = insertBST(r->right, key);
    return r;
}

bool searchBST(BNode* r, int key) {
    if (!r) return false;
    if (key == r->key) return true;
    if (key < r->key) return searchBST(r->left, key);
    return searchBST(r->right, key);
}

BNode* findMin(BNode* r) {
    while (r && r->left) r = r->left;
    return r;
}

BNode* deleteBST(BNode* r, int key) {
    if (!r) return nullptr;
    if (key < r->key) r->left = deleteBST(r->left, key);
    else if (key > r->key) r->right = deleteBST(r->right, key);
    else {
        if (!r->left) { BNode* t = r->right; delete r; return t; }
        if (!r->right) { BNode* t = r->left; delete r; return t; }
        BNode* succ = findMin(r->right); // two children：找 successor
        r->key = succ->key;
        r->right = deleteBST(r->right, succ->key);
    }
    return r;
}

void inorderBST(BNode* r) {
    if (!r) return;
    inorderBST(r->left);
    cout << r->key << ' ';
    inorderBST(r->right);
}

int main() {
    cout << "Binary Tree Traversal:\n";
    TNode* root = new TNode('+');
    root->left = new TNode('*'); root->right = new TNode('E');
    root->left->left = new TNode('A'); root->left->right = new TNode('B');
    cout << "Preorder: "; preorder(root); cout << '\n';
    cout << "Inorder: "; inorder(root); cout << '\n';
    cout << "Postorder: "; postorder(root); cout << "\n\n";

    cout << "Max Heap:\n";
    MaxHeap h;
    for (int x : {20,15,2,14,10,1,21}) h.push(x);
    h.print();
    h.pop();
    cout << "After pop: "; h.print(); cout << '\n';

    cout << "BST:\n";
    BNode* bst = nullptr;
    for (int x : {7,6,10,8,14,12}) bst = insertBST(bst, x);
    cout << "Search 8: " << (searchBST(bst, 8) ? "found" : "not found") << '\n';
    bst = insertBST(bst, 9);
    cout << "After insert 9 inorder: "; inorderBST(bst); cout << '\n';
    bst = deleteBST(bst, 10);
    cout << "After delete 10 inorder: "; inorderBST(bst); cout << '\n';
    return 0;
}

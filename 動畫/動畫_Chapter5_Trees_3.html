#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
using namespace std;

// 一、Binary Tree 節點
struct TreeNode {
    char data;
    TreeNode *left, *right;
    TreeNode(char d) : data(d), left(nullptr), right(nullptr) {}
};

void preorder(TreeNode* p) { // VLR：先訪問，再左，再右
    if (!p) return;
    cout << p->data << ' ';
    preorder(p->left);
    preorder(p->right);
}
void inorder(TreeNode* p) { // LVR：左、訪問、右
    if (!p) return;
    inorder(p->left);
    cout << p->data << ' ';
    inorder(p->right);
}
void postorder(TreeNode* p) { // LRV：左、右、訪問
    if (!p) return;
    postorder(p->left);
    postorder(p->right);
    cout << p->data << ' ';
}

// 二、Max Heap：priority queue 的典型實作
class MaxHeap {
    vector<int> heap; // heap[0] 不用，root 放 heap[1]
public:
    MaxHeap() { heap.push_back(0); }
    bool empty() const { return heap.size() == 1; }
    int top() const { return heap[1]; }
    void push(int x) {
        heap.push_back(x);
        int i = (int)heap.size() - 1;
        while (i > 1 && heap[i/2] < x) {
            heap[i] = heap[i/2]; // parent 往下搬
            i /= 2;
        }
        heap[i] = x;
    }
    void pop() {
        if (empty()) return;
        int last = heap.back(); heap.pop_back();
        if (empty()) return;
        int i = 1, child = 2;
        while (child < (int)heap.size()) {
            if (child + 1 < (int)heap.size() && heap[child] < heap[child+1]) child++;
            if (last >= heap[child]) break;
            heap[i] = heap[child];
            i = child; child *= 2;
        }
        heap[i] = last;
    }
    void print() const {
        for (int i = 1; i < (int)heap.size(); ++i) cout << heap[i] << ' ';
        cout << '\n';
    }
};

// 三、BST：dictionary / dynamic set
struct BSTNode {
    int key;
    BSTNode *left, *right;
    BSTNode(int k): key(k), left(nullptr), right(nullptr) {}
};
BSTNode* insertBST(BSTNode* root, int key) {
    if (!root) return new BSTNode(key);
    if (key < root->key) root->left = insertBST(root->left, key);
    else if (key > root->key) root->right = insertBST(root->right, key);
    return root;
}
bool searchBST(BSTNode* root, int key) {
    while (root) {
        if (key < root->key) root = root->left;
        else if (key > root->key) root = root->right;
        else return true;
    }
    return false;
}

// 四、Disjoint Set：Weighted Union + Path Compression
class DSU {
    vector<int> parent; // root 存負數大小；非 root 存 parent index
public:
    DSU(int n): parent(n, -1) {}
    int find(int x) {
        if (parent[x] < 0) return x;
        return parent[x] = find(parent[x]); // path compression
    }
    void unite(int a, int b) {
        int ra = find(a), rb = find(b);
        if (ra == rb) return;
        if (parent[ra] > parent[rb]) swap(ra, rb); // ra 集合較大，因為負數更小
        parent[ra] += parent[rb];
        parent[rb] = ra;
    }
    void print() const {
        for (int x : parent) cout << x << ' ';
        cout << '\n';
    }
};

int main() {
    TreeNode* root = new TreeNode('A');
    root->left = new TreeNode('B'); root->right = new TreeNode('C');
    root->left->left = new TreeNode('D'); root->left->right = new TreeNode('E');
    cout << "Preorder: "; preorder(root); cout << "\nInorder: "; inorder(root); cout << "\nPostorder: "; postorder(root); cout << "\n\n";

    MaxHeap hp; for (int x : {20,15,2,14,10,1,21}) hp.push(x);
    cout << "MaxHeap after pushes: "; hp.print();
    hp.pop(); cout << "After pop max: "; hp.print();

    BSTNode* bst = nullptr; for (int x : {7,6,8,10,14,12,9}) bst = insertBST(bst, x);
    cout << "Search 9 in BST: " << (searchBST(bst,9)?"found":"not found") << "\n";

    DSU dsu(8); dsu.unite(0,2); dsu.unite(2,5); dsu.unite(5,7);
    cout << "DSU parent array: "; dsu.print();
    cout << "find(7) = " << dsu.find(7) << "\nAfter compression: "; dsu.print();
    return 0;
}

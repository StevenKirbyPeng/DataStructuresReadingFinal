二、演算法總流程

1、流程逐步說明
步驟 1：Read Input
讀入矩形數量、每個矩形的 ID / width / height，以及最後一行 postfix expression。
步驟 2：Build Tree
掃描 expression。遇到數字建立 leaf node 並 push；遇到 V 或 H 就 pop 兩個節點建立 parent node。
步驟 3：DP Merge
每個葉節點保存原方向與旋轉方向；每個 internal node 合併左右子樹的所有可能尺寸。
步驟 4：Prune
刪除被支配的尺寸。例如某狀態寬與高都比另一狀態大，就不可能是最佳解。
步驟 5：Assign Coordinates
從 root 選出最小 area 的狀態後，由上而下回推每個 rectangle 的 (x,y)。

四、V / H 合併與旋轉 DP

1、DP state 的意思

每個節點不是只存一組寬高，而是存多個 ShapeState。例如矩形 1 原本是 (20,10)，旋轉後是 (10,20)。父節點合併時，會嘗試左子樹所有 state × 右子樹所有 state。

struct ShapeState {
    int w = 0, h = 0;
    int leftChoice = -1;
    int rightChoice = -1;
};

其中 leftChoice 與 rightChoice 是為了最後回推座標時，知道最佳解是由哪兩個子狀態組成。

2、Prune 為什麼可以刪狀態？

如果狀態 A 的寬與高都大於或等於狀態 B，且至少一個嚴格較大，A 就被 B 支配。A 之後不可能得到比 B 更好的面積，因此可以安全刪除。

若 B.w ≤ A.w 且 B.h ≤ A.h，則 A 可刪除

#include <algorithm>
#include <cctype>
#include <fstream>
#include <iostream>
#include <limits>
#include <map>
#include <memory>
#include <regex>
#include <sstream>
#include <stack>
#include <string>
#include <vector>

using namespace std;

struct RectInfo {
    int id = 0;
    int w = 0;
    int h = 0;
};

struct Placement {
    int w = 0, h = 0;
    int x = 0, y = 0;
};

struct ShapeState {
    int w = 0, h = 0;
    int leftChoice = -1;
    int rightChoice = -1;
};

class SlicingNode {
public:
    bool isLeaf;
    int rectId;
    char op;
    SlicingNode* left;
    SlicingNode* right;
    vector<ShapeState> states;

    explicit SlicingNode(int id)
        : isLeaf(true), rectId(id), op('\0'), left(nullptr), right(nullptr) {}

    SlicingNode(char cut, SlicingNode* l, SlicingNode* r)
        : isLeaf(false), rectId(-1), op(cut), left(l), right(r) {}
};

class Floorplanner {
private:
    map<int, RectInfo> rects;
    vector<unique_ptr<SlicingNode>> nodes;
    SlicingNode* root = nullptr;
    map<int, Placement> answer;

    static bool isNumberToken(const string& s) {
        if (s.empty()) return false;
        for (char c : s) if (!isdigit(static_cast<unsigned char>(c))) return false;
        return true;
    }

    SlicingNode* makeLeaf(int id) {
        nodes.emplace_back(new SlicingNode(id));
        return nodes.back().get();
    }

    SlicingNode* makeInternal(char op, SlicingNode* left, SlicingNode* right) {
        nodes.emplace_back(new SlicingNode(op, left, right));
        return nodes.back().get();
    }

    static string outputNameFromInput(const string& inputName) {
        regex rgx("benchmark_([0-9]+)\\.txt");
        smatch m;
        if (regex_search(inputName, m, rgx)) return "result_" + m[1].str() + ".txt";
        return "result.txt";
    }

    static void pruneStates(vector<ShapeState>& v) {
        sort(v.begin(), v.end(), [](const ShapeState& a, const ShapeState& b) {
            long long areaA = 1LL * a.w * a.h;
            long long areaB = 1LL * b.w * b.h;
            if (a.w != b.w) return a.w < b.w;
            if (a.h != b.h) return a.h < b.h;
            return areaA < areaB;
        });

        vector<ShapeState> uniqueStates;
        for (const auto& s : v) {
            if (!uniqueStates.empty() && uniqueStates.back().w == s.w && uniqueStates.back().h == s.h) {
                continue;
            }
            uniqueStates.push_back(s);
        }

        vector<ShapeState> pruned;
        for (const auto& s : uniqueStates) {
            bool dominated = false;
            for (const auto& t : uniqueStates) {
                if (&s == &t) continue;
                if (t.w <= s.w && t.h <= s.h && (t.w < s.w || t.h < s.h)) {
                    dominated = true;
                    break;
                }
            }
            if (!dominated) pruned.push_back(s);
        }
        v.swap(pruned);
    }

    void computeStates(SlicingNode* node) {
        if (node->isLeaf) {
            const RectInfo& r = rects[node->rectId];
            node->states.push_back({r.w, r.h, -1, -1});
            if (r.w != r.h) node->states.push_back({r.h, r.w, -1, -1});
            pruneStates(node->states);
            return;
        }

        computeStates(node->left);
        computeStates(node->right);

        for (int i = 0; i < static_cast<int>(node->left->states.size()); ++i) {
            for (int j = 0; j < static_cast<int>(node->right->states.size()); ++j) {
                const ShapeState& a = node->left->states[i];
                const ShapeState& b = node->right->states[j];
                ShapeState merged;
                if (node->op == 'V') {
                    merged.w = a.w + b.w;
                    merged.h = max(a.h, b.h);
                } else {
                    merged.w = max(a.w, b.w);
                    merged.h = a.h + b.h;
                }
                merged.leftChoice = i;
                merged.rightChoice = j;
                node->states.push_back(merged);
            }
        }
        pruneStates(node->states);
    }

    int chooseBestRootState() const {
        int best = 0;
        for (int i = 1; i < static_cast<int>(root->states.size()); ++i) {
            const ShapeState& a = root->states[i];
            const ShapeState& b = root->states[best];
            long long areaA = 1LL * a.w * a.h;
            long long areaB = 1LL * b.w * b.h;
            if (areaA < areaB || (areaA == areaB && (a.w < b.w || (a.w == b.w && a.h < b.h)))) {
                best = i;
            }
        }
        return best;
    }

    void assignCoordinates(SlicingNode* node, int choice, int x, int y) {
        const ShapeState& s = node->states[choice];
        if (node->isLeaf) {
            answer[node->rectId] = {s.w, s.h, x, y};
            return;
        }

        const ShapeState& leftState = node->left->states[s.leftChoice];
        assignCoordinates(node->left, s.leftChoice, x, y);
        if (node->op == 'V') {
            assignCoordinates(node->right, s.rightChoice, x + leftState.w, y);
        } else {
            assignCoordinates(node->right, s.rightChoice, x, y + leftState.h);
        }
    }

    string inorderExpression(SlicingNode* node) const {
        if (node->isLeaf) return to_string(node->rectId);
        return "(" + inorderExpression(node->left) + node->op + inorderExpression(node->right) + ")";
    }

public:
    bool readInput(const string& inputName) {
        ifstream fin(inputName);
        if (!fin) return false;

        string tag;
        int n;
        fin >> tag >> n;
        for (int i = 0; i < n; ++i) {
            RectInfo r;
            fin >> r.id >> r.w >> r.h;
            rects[r.id] = r;
        }

        fin >> tag; // Expression
        vector<string> expr;
        string token;
        while (fin >> token) expr.push_back(token);

        stack<SlicingNode*> st;
        for (const string& tok : expr) {
            if (isNumberToken(tok)) {
                st.push(makeLeaf(stoi(tok)));
            } else if (tok == "V" || tok == "H") {
                if (st.size() < 2) return false;
                SlicingNode* right = st.top(); st.pop();
                SlicingNode* left = st.top(); st.pop();
                st.push(makeInternal(tok[0], left, right));
            }
        }
        if (st.size() != 1) return false;
        root = st.top();
        return true;
    }

    bool solveAndWrite(const string& inputName) {
        if (!root) return false;
        computeStates(root);
        int best = chooseBestRootState();
        assignCoordinates(root, best, 0, 0);

        const ShapeState& finalState = root->states[best];
        string outName = outputNameFromInput(inputName);
        ofstream fout(outName);
        if (!fout) return false;

        fout << "In-order: " << inorderExpression(root) << '\n';
        fout << "Total width: " << finalState.w << '\n';
        fout << "Total height: " << finalState.h << '\n';
        fout << "Minimum area: " << 1LL * finalState.w * finalState.h << '\n';
        for (const auto& item : rects) {
            int id = item.first;
            const Placement& p = answer[id];
            fout << id << ": (" << p.w << ", " << p.h << ") at (" << p.x << ", " << p.y << ")\n";
        }
        return true;
    }
};

int main(int argc, char* argv[]) {
    if (argc < 2) {
        cerr << "Usage: " << argv[0] << " benchmark_X.txt\n";
        return 1;
    }

    string inputName = argv[1];
    Floorplanner planner;
    if (!planner.readInput(inputName)) {
        cerr << "Error: cannot read or parse input file.\n";
        return 1;
    }
    if (!planner.solveAndWrite(inputName)) {
        cerr << "Error: cannot write result file.\n";
        return 1;
    }
    return 0;
}



1、程式段落逐步講解


（一）輸入與輸出檔名
argv[1] 是執行時輸入的檔名，例如 benchmark_1.txt。程式再用 regex 抓出數字，產生 result_1.txt。


（二）Stack 建樹
Postfix expression 本身就是 slicing tree 的 postorder traversal，所以遇到 operator 時，stack 內最後兩個 node 就是 operator 的左右子樹。


（三）DP 與旋轉
每個 leaf node 先保存原尺寸與旋轉尺寸。internal node 根據 V/H 公式合併子狀態，再 pruning。


（四）最佳解與座標
root 的每個 state 都代表一種完整 floorplan 尺寸。程式選出面積最小者，再由 root 往下回推每個 rectangle 的座標。

/*
 * 資料結構 Homework #3：Slicing Tree Floorplanning
 * --------------------------------------------------
 * 本程式功能：
 * 1. 從 benchmark_X.txt 讀取矩形資料與 postfix slicing expression。

 * 4. 由下而上計算每個 slicing tree 節點可能形成的外框尺寸。
 * 5. 刪除被支配的尺寸狀態，保留可能成為最佳解的候選尺寸。
 * 6. 在 root 選出最小面積的尺寸，並由上而下回推每個矩形座標。
 * 7. 輸出 result_X.txt，包含中序表示式、總寬高、最小面積與每個矩形的位置。
 *
 * 注意：
 * - 以下只加入中文註解，沒有改變任何原始程式邏輯、變數、函式或輸出格式。
 * - HW3 的核心資料結構是 stack 與 binary tree。
 */

#include <algorithm>  // 提供 sort、max 等演算法函式

#include <memory>     // 提供 unique_ptr，用來自動管理節點記憶體

#include <sstream>    // 提供字串串流工具，本程式中保留 include 不影響功能

#include <string>     // 提供 string

using namespace std;

// RectInfo：儲存輸入檔中每個基本矩形的資訊。
// id：矩形編號。
// w ：原始寬度。
// h ：原始高度。
struct RectInfo {
    int id = 0;
    int w = 0;
    int h = 0;
};

// Placement：儲存最後輸出時，每個矩形的實際擺放結果。
// w, h：最後決定後的寬與高，可能是原始方向，也可能是旋轉後方向。
// x, y：矩形左下角座標。
struct Placement {
    int w = 0, h = 0;
    int x = 0, y = 0;
};

// ShapeState：表示某個 slicing tree 節點可能形成的一種外框尺寸。
// 對 leaf 來說，狀態代表該矩形不旋轉或旋轉後的尺寸。
// 對 internal node 來說，狀態代表左右子樹經由 H/V cut 合併後的外框尺寸。
// leftChoice/rightChoice：記錄此狀態是由左右子節點的哪一個狀態合併而來，
// 方便最後由 root 回推每個矩形的座標與旋轉方向。
struct ShapeState {
    int w = 0, h = 0;
    int leftChoice = -1;
    int rightChoice = -1;
};

// SlicingNode：slicing tree 的節點。
// leaf node 代表一個基本矩形。
// internal node 代表一個切割操作，op 只會是 'V' 或 'H'。
class SlicingNode {
public:
    bool isLeaf;                 // true 表示 leaf；false 表示 internal node
    int rectId;                  // leaf 使用：矩形 id；internal node 設為 -1
    char op;                     // internal node 使用：'V' 或 'H'；leaf 設為 '\0'
    SlicingNode* left;           // 左子樹
    SlicingNode* right;          // 右子樹
    vector<ShapeState> states;   // 此節點所有尚未被刪除的候選外框狀態

    // 建立 leaf node：只需要矩形 id。
    explicit SlicingNode(int id)
        : isLeaf(true), rectId(id), op('\0'), left(nullptr), right(nullptr) {}

    // 建立 internal node：需要 cut 操作與左右子節點。
    SlicingNode(char cut, SlicingNode* l, SlicingNode* r)
        : isLeaf(false), rectId(-1), op(cut), left(l), right(r) {}
};

// Floorplanner：封裝整個 HW3 floorplanning 流程。
// 包含讀檔、建樹、計算最佳尺寸、回推座標、寫檔。
class Floorplanner {
private:
    map<int, RectInfo> rects;              // 由矩形 id 對應到原始矩形資訊
    vector<unique_ptr<SlicingNode>> nodes; // 管理所有 new 出來的 slicing tree 節點
    SlicingNode* root = nullptr;           // slicing tree 的根節點
    map<int, Placement> answer;            // 每個矩形最後的尺寸與座標

    // 判斷 token 是否為純數字。
    // 在 postfix expression 中，數字代表矩形 id；V/H 代表切割操作。
    static bool isNumberToken(const string& s) {
        if (s.empty()) return false;
        for (char c : s) if (!isdigit(static_cast<unsigned char>(c))) return false;
        return true;
    }

    // 建立 leaf node，並交給 unique_ptr 管理生命週期。
    SlicingNode* makeLeaf(int id) {
        nodes.emplace_back(new SlicingNode(id));
        return nodes.back().get();
    }

    // 建立 internal node，並交給 unique_ptr 管理生命週期。
    SlicingNode* makeInternal(char op, SlicingNode* left, SlicingNode* right) {
        nodes.emplace_back(new SlicingNode(op, left, right));
        return nodes.back().get();
    }

    // 根據輸入檔名 benchmark_X.txt 產生輸出檔名 result_X.txt。
    // 若檔名不符合 benchmark_X.txt 格式，就輸出 result.txt。
    static string outputNameFromInput(const string& inputName) {
        regex rgx("benchmark_([0-9]+)\\.txt");
        smatch m;
        if (regex_search(inputName, m, rgx)) return "result_" + m[1].str() + ".txt";
        return "result.txt";
    }

    // pruneStates：刪除不必要的候選狀態。
    // 若狀態 A 的寬高都不比狀態 B 好，且至少一邊更差，則 A 被 B dominated。
    // 被 dominated 的狀態不可能產生更小外框，因此可以移除以降低後續合併數量。
    static void pruneStates(vector<ShapeState>& v) {
        // 先排序，讓相同 (w, h) 的狀態靠在一起，方便去除重複尺寸。
        sort(v.begin(), v.end(), [](const ShapeState& a, const ShapeState& b) {
            long long areaA = 1LL * a.w * a.h;
            long long areaB = 1LL * b.w * b.h;
            if (a.w != b.w) return a.w < b.w;
            if (a.h != b.h) return a.h < b.h;
            return areaA < areaB;
        });

        // 移除完全相同寬高的重複狀態。
        vector<ShapeState> uniqueStates;
        for (const auto& s : v) {
            if (!uniqueStates.empty() && uniqueStates.back().w == s.w && uniqueStates.back().h == s.h) {
                continue;
            }
            uniqueStates.push_back(s);
        }

        // 移除 dominated state。
        // t.w <= s.w 且 t.h <= s.h，並且至少一邊嚴格較小，表示 s 永遠不會比 t 好。
        vector<ShapeState> pruned;
        for (const auto& s : uniqueStates) {
            bool dominated = false;
            for (const auto& t : uniqueStates) {
                if (&s == &t) continue;
                if (t.w <= s.w && t.h <= s.h && (t.w < s.w || t.h < s.h)) {
                    dominated = true;
                    break;
                }
            }
            if (!dominated) pruned.push_back(s);
        }
        v.swap(pruned);
    }

    // computeStates：由下而上計算每個節點的所有可能外框尺寸。
    // leaf：產生原方向與旋轉方向。
    // internal node：先算左右子樹，再依照 V/H cut 合併左右所有候選狀態。
    void computeStates(SlicingNode* node) {
        // leaf node：一個矩形有最多兩種方向：(w,h) 與 (h,w)。
        if (node->isLeaf) {
            const RectInfo& r = rects[node->rectId];
            node->states.push_back({r.w, r.h, -1, -1});
            if (r.w != r.h) node->states.push_back({r.h, r.w, -1, -1});
            pruneStates(node->states);
            return;
        }

        // internal node：必須先知道左右子樹各自有哪些候選外框。
        computeStates(node->left);
        computeStates(node->right);

        // 枚舉左子樹狀態 i 與右子樹狀態 j 的所有組合。
        for (int i = 0; i < static_cast<int>(node->left->states.size()); ++i) {
            for (int j = 0; j < static_cast<int>(node->right->states.size()); ++j) {
                const ShapeState& a = node->left->states[i];
                const ShapeState& b = node->right->states[j];
                ShapeState merged;

                // V cut：左子樹在左、右子樹在右。
                // 合併後寬度相加，高度取最大。
                if (node->op == 'V') {
                    merged.w = a.w + b.w;
                    merged.h = max(a.h, b.h);
                } else {
                    // H cut：左子樹在下、右子樹在上。
                    // 合併後寬度取最大，高度相加。
                    merged.w = max(a.w, b.w);
                    merged.h = a.h + b.h;
                }

                // 記錄此合併狀態來自左右子樹哪個狀態，之後 assignCoordinates 會用到。
                merged.leftChoice = i;
                merged.rightChoice = j;
                node->states.push_back(merged);
            }
        }

        // 合併後刪除不可能成為最佳解的狀態。
        pruneStates(node->states);
    }

    // chooseBestRootState：在 root 的候選狀態中選擇最終答案。
    // 主要比較面積；若面積相同，則依序比較寬度與高度，讓輸出固定且穩定。
    int chooseBestRootState() const {
        int best = 0;
        for (int i = 1; i < static_cast<int>(root->states.size()); ++i) {
            const ShapeState& a = root->states[i];
            const ShapeState& b = root->states[best];
            long long areaA = 1LL * a.w * a.h;
            long long areaB = 1LL * b.w * b.h;
            if (areaA < areaB || (areaA == areaB && (a.w < b.w || (a.w == b.w && a.h < b.h)))) {
                best = i;
            }
        }
        return best;
    }

    // assignCoordinates：由 root 選定的狀態往下回推每個矩形座標。
    // x, y 表示目前 node 外框左下角座標。
    void assignCoordinates(SlicingNode* node, int choice, int x, int y) {
        const ShapeState& s = node->states[choice];

        // 若是 leaf，代表已經到達一個基本矩形，直接記錄答案。
        if (node->isLeaf) {
            answer[node->rectId] = {s.w, s.h, x, y};
            return;
        }

        // 取得左子樹被選中的尺寸，因為右子樹的位置會依賴左子樹寬/高。
        const ShapeState& leftState = node->left->states[s.leftChoice];

        // 左子樹永遠放在目前外框的左下角。
        assignCoordinates(node->left, s.leftChoice, x, y);

        // V cut：右子樹放在左子樹右邊，所以 x 需要加上左子樹寬度。
        if (node->op == 'V') {
            assignCoordinates(node->right, s.rightChoice, x + leftState.w, y);
        } else {
            // H cut：右子樹放在左子樹上方，所以 y 需要加上左子樹高度。
            assignCoordinates(node->right, s.rightChoice, x, y + leftState.h);
        }
    }

    // inorderExpression：輸出有括號的中序 slicing expression。
    // leaf 直接輸出矩形 id；internal node 輸出 (left op right)。
    string inorderExpression(SlicingNode* node) const {
        if (node->isLeaf) return to_string(node->rectId);
        return "(" + inorderExpression(node->left) + node->op + inorderExpression(node->right) + ")";
    }

public:
    // readInput：讀取 benchmark 檔，並使用 postfix expression 建立 slicing tree。
    bool readInput(const string& inputName) {
        ifstream fin(inputName);
        if (!fin) return false;

        string tag;
        int n;

        // 讀取 NumRectangles n。
        fin >> tag >> n;

        // 讀取 n 個矩形的 id、width、height。
        for (int i = 0; i < n; ++i) {
            RectInfo r;
            fin >> r.id >> r.w >> r.h;
            rects[r.id] = r;
        }

        // 讀取 Expression 後面的 postfix tokens。
        fin >> tag; // Expression
        vector<string> expr;
        string token;
        while (fin >> token) expr.push_back(token);

        // 利用 stack 建立 slicing tree。
        // 遇到矩形 id：建立 leaf 並 push。
        // 遇到 V/H：pop 出右、左子樹，建立 internal node 後再 push 回去。
        stack<SlicingNode*> st;
        for (const string& tok : expr) {
            if (isNumberToken(tok)) {
                st.push(makeLeaf(stoi(tok)));
            } else if (tok == "V" || tok == "H") {
                if (st.size() < 2) return false;
                SlicingNode* right = st.top(); st.pop();
                SlicingNode* left = st.top(); st.pop();
                st.push(makeInternal(tok[0], left, right));
            }
        }

        // 正確的 postfix expression 最後 stack 應該只剩一個節點，也就是 root。
        if (st.size() != 1) return false;
        root = st.top();
        return true;
    }

    // solveAndWrite：執行完整求解流程並寫出結果檔。
    bool solveAndWrite(const string& inputName) {
        if (!root) return false;

        // 1. 動態規劃：計算 root 以下所有節點可能尺寸。
        computeStates(root);

        // 2. 從 root 候選狀態選出面積最小者。
        int best = chooseBestRootState();

        // 3. 從 root 左下角 (0,0) 開始回推每個矩形座標。
        assignCoordinates(root, best, 0, 0);

        // 4. 取得最終總寬高與輸出檔名。
        const ShapeState& finalState = root->states[best];
        string outName = outputNameFromInput(inputName);
        ofstream fout(outName);
        if (!fout) return false;

        // 5. 依照作業要求格式輸出結果。
        fout << "In-order: " << inorderExpression(root) << '\n';
        fout << "Total width: " << finalState.w << '\n';
        fout << "Total height: " << finalState.h << '\n';
        fout << "Minimum area: " << 1LL * finalState.w * finalState.h << '\n';
        for (const auto& item : rects) {
            int id = item.first;
            const Placement& p = answer[id];
            fout << id << ": (" << p.w << ", " << p.h << ") at (" << p.x << ", " << p.y << ")\n";
        }
        return true;
    }
};

// main：程式進入點。
// 使用方式：./E24134778 benchmark_X.txt
int main(int argc, char* argv[]) {
    // 若沒有提供輸入檔名，印出使用方式並結束。
    if (argc < 2) {
        cerr << "Usage: " << argv[0] << " benchmark_X.txt\n";
        return 1;
    }

    // argv[1] 是輸入檔名，例如 benchmark_0.txt。
    string inputName = argv[1];

    // 建立 Floorplanner 物件，負責完整求解流程。
    Floorplanner planner;

    // 讀檔與建立 slicing tree。
    if (!planner.readInput(inputName)) {
        cerr << "Error: cannot read or parse input file.\n";
        return 1;
    }

    // 計算最佳 floorplan 並輸出 result_X.txt。
    if (!planner.solveAndWrite(inputName)) {
        cerr << "Error: cannot write result file.\n";
        return 1;
    }

    // 正常結束。
    return 0;
}







  
八、程式流程圖

  
以下流程圖說明 HW3 程式從讀取 benchmark、建立 postfix slicing tree、DP 合併候選尺寸、回推座標，到輸出 result_X.txt 的完整流程。

  




  
流程圖逐步說明

    
1、main() 先從命令列取得 benchmark_X.txt，符合助教執行方式。

    
2、readInput() 使用 ifstream 讀入矩形資料與 postfix expression。

    
3、遇到數字 token 時建立 leaf node 並 push 進 stack；遇到 V 或 H 時 pop 兩棵子樹建立 internal node。

    
4、computeStates() 對每個節點計算所有可能外框尺寸。leaf 會考慮旋轉，internal node 會依照 V/H 合併公式產生新狀態。

    
5、pruneStates() 移除被支配的尺寸狀態，避免狀態數過多。

    
6、chooseBestRootState() 從 root 的候選狀態中選出最小面積。

    
7、assignCoordinates() 從 root 往 leaf 回推每個矩形的左下角座標。

    
8、最後輸出中序式、總寬、總高、最小面積與每個 rectangle 的尺寸與座標。

  






  
九、每個函式與資料結構說明

  

    
名稱類型功能說明在 HW3 的角色

    
      
RectInfostruct保存輸入矩形的 id、原始寬度與高度。作為 leaf node 產生尺寸狀態的依據。

      
Placementstruct保存最後輸出的寬高與左下角座標。最後寫入 result_X.txt。

      
ShapeStatestruct保存某個節點的一種候選外框尺寸，並記錄左右子節點選用的 state index。DP 狀態與座標回推的核心。

      
SlicingNodeclass表示 slicing tree 節點，leaf 是 rectangle，internal node 是 V/H cut。建立二元切割樹。

      
isNumberToken()static function判斷 postfix token 是否為數字。區分 rectangle id 與 V/H operator。

      
makeLeaf()member function建立 leaf node，並用 unique_ptr 管理記憶體。讀到數字 token 時使用。

      
makeInternal()member function建立 V/H internal node，接上左右子樹。讀到 V/H token 時使用。

      
outputNameFromInput()static function用 regex 將 benchmark_X.txt 轉成 result_X.txt。符合助教輸出檔名要求。

      
pruneStates()static function刪除重複尺寸與 dominated state。降低 DP 狀態數，提高效率。

      
computeStates()member function遞迴計算每個節點的候選外框尺寸。找最小面積前的核心 DP 過程。

      
chooseBestRootState()member function從 root 的 states 中挑出面積最小的候選解。決定最終 floorplan 尺寸。

      
assignCoordinates()member function由 root 往下依照選定 state 回推每個矩形座標。產生 final placement。

      
inorderExpression()member function遞迴產生有括號、無空格的中序表示式。輸出 In-order: 那一行。

      
readInput()public member function讀取 benchmark，並用 stack 建立 slicing tree。程式輸入與建樹核心。

      
solveAndWrite()public member function執行 DP、選最佳解、回推座標、寫出 result_X.txt。整個求解與輸出流程。

      
main()function處理命令列參數，呼叫 readInput 與 solveAndWrite。符合助教編譯與執行規格。

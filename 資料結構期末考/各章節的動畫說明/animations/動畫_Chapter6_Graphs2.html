#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
#include <tuple>
using namespace std;
const int INF = 1e9;

class DSU {
    vector<int> parent, rankv;
public:
    DSU(int n): parent(n), rankv(n,0) { for(int i=0;i<n;i++) parent[i]=i; }
    int find(int x){ return parent[x]==x ? x : parent[x]=find(parent[x]); }
    bool unite(int a,int b){
        a=find(a); b=find(b);
        if(a==b) return false;
        if(rankv[a]<rankv[b]) swap(a,b);
        parent[b]=a;
        if(rankv[a]==rankv[b]) rankv[a]++;
        return true;
    }
};

class Graph {
    int n;
    vector<vector<pair<int,int>>> adj;
    vector<tuple<int,int,int>> edges;
    void dfsUtil(int u, vector<bool>& vis){
        vis[u]=true; cout<<u<<' ';
        for(auto [v,w]: adj[u]) if(!vis[v]) dfsUtil(v,vis);
    }
public:
    Graph(int n): n(n), adj(n) {}
    void addEdge(int u,int v,int w=1){
        adj[u].push_back({v,w});
        adj[v].push_back({u,w});
        edges.push_back({w,u,v});
    }
    void dfs(int s){ vector<bool> vis(n,false); cout<<"DFS: "; dfsUtil(s,vis); cout<<'\n'; }
    void bfs(int s){
        vector<bool> vis(n,false); queue<int> q;
        vis[s]=true; q.push(s); cout<<"BFS: ";
        while(!q.empty()){
            int u=q.front(); q.pop(); cout<<u<<' ';
            for(auto [v,w]: adj[u]) if(!vis[v]){vis[v]=true; q.push(v);}
        }
        cout<<'\n';
    }
    void dijkstra(int s){
        vector<int> dist(n,INF); vector<bool> done(n,false); dist[s]=0;
        for(int r=0;r<n;r++){
            int u=-1;
            for(int i=0;i<n;i++) if(!done[i] && (u==-1 || dist[i]<dist[u])) u=i;
            if(u==-1 || dist[u]==INF) break;
            done[u]=true;
            for(auto [v,w]: adj[u]) if(!done[v] && dist[u]+w<dist[v]) dist[v]=dist[u]+w;
        }
        cout<<"Dijkstra: "; for(int d: dist) cout<<d<<' '; cout<<'\n';
    }
    void kruskal(){
        sort(edges.begin(), edges.end()); DSU dsu(n); int total=0;
        cout<<"Kruskal MST:\n";
        for(auto [w,u,v]: edges) if(dsu.unite(u,v)){
            cout<<"  choose ("<<u<<","<<v<<") w="<<w<<'\n'; total+=w;
        }
        cout<<"total="<<total<<'\n';
    }
};

void floyd(vector<vector<int>> d){
    int n=d.size();
    for(int k=0;k<n;k++) for(int i=0;i<n;i++) for(int j=0;j<n;j++)
        if(d[i][k]<INF && d[k][j]<INF) d[i][j]=min(d[i][j], d[i][k]+d[k][j]);
    cout<<"Floyd matrix:\n";
    for(auto &row:d){ for(int x:row) cout<<(x==INF?-1:x)<<'\t'; cout<<'\n'; }
}

vector<int> topoSort(int n, vector<vector<int>> dag){
    vector<int> indeg(n), ans; queue<int> q;
    for(int u=0;u<n;u++) for(int v:dag[u]) indeg[v]++;
    for(int i=0;i<n;i++) if(indeg[i]==0) q.push(i);
    while(!q.empty()){
        int u=q.front(); q.pop(); ans.push_back(u);
        for(int v:dag[u]) if(--indeg[v]==0) q.push(v);
    }
    return ans;
}

int main(){
    Graph g(6);
    g.addEdge(0,1,7); g.addEdge(0,2,9); g.addEdge(0,5,14);
    g.addEdge(1,2,10); g.addEdge(1,3,15); g.addEdge(2,3,11);
    g.addEdge(2,5,2); g.addEdge(3,4,6); g.addEdge(4,5,9);
    g.dfs(0); g.bfs(0); g.dijkstra(0); g.kruskal();
    floyd({{0,4,INF,6},{INF,0,2,INF},{3,INF,0,1},{INF,INF,INF,0}});
    auto order = topoSort(6, {{1,2},{3},{3,4},{5},{5},{}});
    cout<<"Topo: "; for(int x:order) cout<<x<<' '; cout<<'\n';
}

Chapter 6 Graphs 進階主題完整 C++ 程式
#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
#include <limits>
using namespace std;

const int INF = 1000000000;

struct Edge {
    int u, v, w;
};

// Bellman-Ford：可處理負權重邊，並偵測負環
bool bellmanFord(int n, const vector<Edge>& edges, int s, vector<int>& dist) {
    dist.assign(n, INF);
    dist[s] = 0;
    for (int round = 1; round <= n - 1; ++round) {
        bool changed = false;
        for (auto e : edges) {
            if (dist[e.u] != INF && dist[e.u] + e.w < dist[e.v]) {
                dist[e.v] = dist[e.u] + e.w;
                changed = true;
            }
        }
        if (!changed) break;
    }
    for (auto e : edges) {
        if (dist[e.u] != INF && dist[e.u] + e.w < dist[e.v]) return false;
    }
    return true;
}

// Floyd-Warshall：所有點對最短路徑
void floydWarshall(vector<vector<int>>& d) {
    int n = d.size();
    for (int k = 0; k < n; ++k)
        for (int i = 0; i < n; ++i)
            for (int j = 0; j < n; ++j)
                if (d[i][k] != INF && d[k][j] != INF)
                    d[i][j] = min(d[i][j], d[i][k] + d[k][j]);
}

// Warshall：傳遞閉包，判斷 i 是否可到 j
void transitiveClosure(vector<vector<int>>& reach) {
    int n = reach.size();
    for (int k = 0; k < n; ++k)
        for (int i = 0; i < n; ++i)
            for (int j = 0; j < n; ++j)
                reach[i][j] = reach[i][j] || (reach[i][k] && reach[k][j]);
}

// Topological Sort：Kahn 演算法
vector<int> topologicalSort(int n, vector<vector<int>> adj) {
    vector<int> indeg(n, 0), order;
    for (int u = 0; u < n; ++u)
        for (int v : adj[u]) indeg[v]++;
    queue<int> q;
    for (int i = 0; i < n; ++i) if (indeg[i] == 0) q.push(i);
    while (!q.empty()) {
        int u = q.front(); q.pop();
        order.push_back(u);
        for (int v : adj[u]) if (--indeg[v] == 0) q.push(v);
    }
    return order;
}

// Critical Path：AOE network
void criticalPath(int n, vector<vector<pair<int,int>>> adj) {
    vector<int> indeg(n,0), topo, ee(n,0), le(n,0);
    for(int u=0;u<n;++u) for(auto [v,w]:adj[u]) indeg[v]++;
    queue<int> q;
    for(int i=0;i<n;++i) if(indeg[i]==0) q.push(i);
    while(!q.empty()){
        int u=q.front(); q.pop(); topo.push_back(u);
        for(auto [v,w]:adj[u]){
            ee[v]=max(ee[v], ee[u]+w);
            if(--indeg[v]==0) q.push(v);
        }
    }
    fill(le.begin(), le.end(), ee[topo.back()]);
    for(int i=(int)topo.size()-1;i>=0;--i){
        int u=topo[i];
        for(auto [v,w]:adj[u]) le[u]=min(le[u], le[v]-w);
    }
    cout << "Critical activities:\n";
    for(int u=0;u<n;++u){
        for(auto [v,w]:adj[u]){
            int e=ee[u];
            int l=le[v]-w;
            if(e==l) cout << u << " -> " << v << " duration=" << w << "\n";
        }
    }
}

int main(){
    vector<Edge> edges={{0,1,6},{0,2,5},{1,3,-1},{2,1,-2},{2,3,4}};
    vector<int> dist;
    if(bellmanFord(4,edges,0,dist)){
        cout << "Bellman-Ford distances from 0: ";
        for(int x:dist) cout << x << ' ';
        cout << "\n";
    } else cout << "Negative cycle detected\n";

    vector<vector<int>> adj={{1,2},{3},{3},{}};
    auto order=topologicalSort(4,adj);
    cout << "Topological order: ";
    for(int x:order) cout << x << ' ';
    cout << "\n";

    vector<vector<pair<int,int>>> aoe(5);
    aoe[0]={{1,3},{2,2}}; aoe[1]={{3,4}}; aoe[2]={{3,1}}; aoe[3]={{4,2}};
    criticalPath(5,aoe);
    return 0;
}

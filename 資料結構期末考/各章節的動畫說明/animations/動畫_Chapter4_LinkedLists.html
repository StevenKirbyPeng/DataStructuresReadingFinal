錯誤細節檢查：
  • 插入節點時，必須先讓新節點接住後半段，再改前一個節點的 link。
  • 刪除節點時，必須先保存要刪的節點位置，再改指標，最後 delete。
  • 反轉串列時，必須先保存 next，否則改掉 current->link 後會找不到後面節點。
  • Queue 變空時，front 與 rear 都要設回 nullptr。

    4. Linked Queue：push、pop、front

// ============================================================
// 一、Singly Linked List 單向鏈結串列
// ============================================================
class SinglyLinkedList {
private:
    struct Node {
        int data;       // 節點資料
        Node* link;     // 指向下一個節點的指標

        Node(int value, Node* next = nullptr)
            : data(value), link(next) {}
    };

    Node* first;        // 指向第一個節點，也就是整條串列的入口

public:
    // 建構子：建立空串列
    SinglyLinkedList() : first(nullptr) {}

    // 解構子：釋放所有節點，避免 memory leak
    ~SinglyLinkedList() {
        clear();
    }

    // 功能：清空整條串列
    // 參數：無
    // 回傳值：無
    void clear() {
        Node* current = first;
        while (current != nullptr) {
            Node* next = current->link;  // 先保存下一個節點
            delete current;              // 再刪除目前節點
            current = next;              // 移動到下一個節點
        }
        first = nullptr;
    }

    // 功能：在串列前端插入
    // 參數：value，要插入的資料
    // 回傳值：無
    // 時間複雜度：O(1)
    void push_front(int value) {
        first = new Node(value, first);
    }

    // 功能：在串列尾端插入
    // 參數：value，要插入的資料
    // 回傳值：無
    // 時間複雜度：O(n)
    void push_back(int value) {
        Node* newNode = new Node(value);

        if (first == nullptr) {
            first = newNode;
            return;
        }

        Node* current = first;
        while (current->link != nullptr) {
            current = current->link;
        }
        current->link = newNode;
    }

    // 功能：在 target 後面插入 value
    // 參數：target，指定節點資料；value，要插入的資料
    // 回傳值：true 表示成功，false 表示找不到 target
    // 正確指標順序：newNode->link = current->link，再 current->link = newNode
    bool insert_after(int target, int value) {
        Node* current = first;

        while (current != nullptr && current->data != target) {
            current = current->link;
        }

        if (current == nullptr) return false;

        Node* newNode = new Node(value);
        newNode->link = current->link;   // 先接住後半段
        current->link = newNode;         // 再讓 target 接到新節點
        return true;
    }

    // 功能：刪除第一個資料等於 value 的節點
    // 參數：value，要刪除的資料
    // 回傳值：true 表示成功，false 表示找不到
    bool remove(int value) {
        if (first == nullptr) return false;

        // Case 1：刪除第一個節點
        if (first->data == value) {
            Node* delNode = first;
            first = first->link;
            delete delNode;
            return true;
        }

        // Case 2：刪除中間或最後節點，需要 prev
        Node* prev = first;
        Node* current = first->link;

        while (current != nullptr && current->data != value) {
            prev = current;
            current = current->link;
        }

        if (current == nullptr) return false;

        prev->link = current->link;      // 前一個節點跳過 current
        delete current;                  // 釋放 current
        return true;
    }

    // 功能：搜尋 value 是否存在
    // 參數：value，要搜尋的資料
    // 回傳值：true 表示找到，false 表示不存在
    bool search(int value) const {
        Node* current = first;
        while (current != nullptr) {
            if (current->data == value) return true;
            current = current->link;
        }
        return false;
    }

    // 功能：原地反轉串列
    // 參數：無
    // 回傳值：無
    // 關鍵：一定要先保存 next，再改 current->link
    void reverse() {
        Node* previous = nullptr;
        Node* current = first;

        while (current != nullptr) {
            Node* next = current->link;  // Step 1：保存下一個節點
            current->link = previous;    // Step 2：反轉目前 link
            previous = current;          // Step 3：previous 往前
            current = next;              // Step 4：current 往前
        }

        first = previous;
    }

    // 功能：印出串列
    void print() const {
        Node* current = first;
        while (current != nullptr) {
            cout << current->data << " -> ";
            current = current->link;
        }
        cout << "NULL\n";
    }
};

// ============================================================
// 二、Doubly Linked List 雙向鏈結串列
// ============================================================
class DoublyLinkedList {
private:
    struct Node {
        int data;
        Node* left;     // 指向前一個節點
        Node* right;    // 指向下一個節點

        Node(int value) : data(value), left(nullptr), right(nullptr) {}
    };

    Node* head;
    Node* tail;

public:
    DoublyLinkedList() : head(nullptr), tail(nullptr) {}

    ~DoublyLinkedList() {
        Node* current = head;
        while (current != nullptr) {
            Node* next = current->right;
            delete current;
            current = next;
        }
    }

    // 功能：在尾端插入
    // 時間複雜度：O(1)，因為有 tail
    void push_back(int value) {
        Node* newNode = new Node(value);

        if (head == nullptr) {
            head = tail = newNode;
            return;
        }

        tail->right = newNode;
        newNode->left = tail;
        tail = newNode;
    }

    // 功能：在 target 後插入 value
    bool insert_after(int target, int value) {
        Node* current = head;

        while (current != nullptr && current->data != target) {
            current = current->right;
        }

        if (current == nullptr) return false;

        Node* newNode = new Node(value);
        Node* nextNode = current->right;

        newNode->left = current;
        newNode->right = nextNode;
        current->right = newNode;

        if (nextNode != nullptr) {
            nextNode->left = newNode;
        } else {
            tail = newNode;
        }

        return true;
    }

    // 功能：刪除第一個資料等於 value 的節點
    bool remove(int value) {
        Node* current = head;

        while (current != nullptr && current->data != value) {
            current = current->right;
        }

        if (current == nullptr) return false;

        if (current->left != nullptr)
            current->left->right = current->right;
        else
            head = current->right;

        if (current->right != nullptr)
            current->right->left = current->left;
        else
            tail = current->left;

        delete current;
        return true;
    }

    void print_forward() const {
        Node* current = head;
        while (current != nullptr) {
            cout << current->data << " <-> ";
            current = current->right;
        }
        cout << "NULL\n";
    }

    void print_backward() const {
        Node* current = tail;
        while (current != nullptr) {
            cout << current->data << " <-> ";
            current = current->left;
        }
        cout << "NULL\n";
    }
};

// ============================================================
// 三、Linked Stack 鏈結堆疊
// ============================================================
class LinkedStack {
private:
    struct Node {
        int data;
        Node* link;
        Node(int value, Node* next = nullptr) : data(value), link(next) {}
    };

    Node* topNode;

public:
    LinkedStack() : topNode(nullptr) {}

    ~LinkedStack() {
        while (!isEmpty()) pop();
    }

    bool isEmpty() const {
        return topNode == nullptr;
    }

    // 功能：Push，把新節點放到 top
    // 時間複雜度：O(1)
    void push(int value) {
        topNode = new Node(value, topNode);
    }

    // 功能：Pop，刪除 top 節點
    // 時間複雜度：O(1)
    void pop() {
        if (isEmpty()) throw runtime_error("Stack is empty");

        Node* delNode = topNode;
        topNode = topNode->link;
        delete delNode;
    }

    int top() const {
        if (isEmpty()) throw runtime_error("Stack is empty");
        return topNode->data;
    }
};

// ============================================================
// 四、Linked Queue 鏈結佇列
// ============================================================
class LinkedQueue {
private:
    struct Node {
        int data;
        Node* link;
        Node(int value, Node* next = nullptr) : data(value), link(next) {}
    };

    Node* frontNode;
    Node* rearNode;

public:
    LinkedQueue() : frontNode(nullptr), rearNode(nullptr) {}

    ~LinkedQueue() {
        while (!isEmpty()) pop();
    }

    bool isEmpty() const {
        return frontNode == nullptr;
    }

    // 功能：Push / Enqueue，把新節點加到 rear
    // 時間複雜度：O(1)
    void push(int value) {
        Node* newNode = new Node(value);

        if (isEmpty()) {
            frontNode = rearNode = newNode;
        } else {
            rearNode->link = newNode;
            rearNode = newNode;
        }
    }

    // 功能：Pop / Dequeue，刪除 front 節點
    // 時間複雜度：O(1)
    void pop() {
        if (isEmpty()) throw runtime_error("Queue is empty");

        Node* delNode = frontNode;
        frontNode = frontNode->link;

        if (frontNode == nullptr) {
            rearNode = nullptr; // 佇列變空時，rear 也要歸零
        }

        delete delNode;
    }

    int front() const {
        if (isEmpty()) throw runtime_error("Queue is empty");
        return frontNode->data;
    }
};

// 五、主程式：測試所有 Chapter 4 Linked List 操作

int main() {
    cout << "===== Singly Linked List 測試 =====\n";
    SinglyLinkedList list;
    list.push_back(10);
    list.push_back(20);
    list.push_back(30);
    cout << "原始串列：";
    list.print();

    list.push_front(5);
    cout << "push_front(5)：";
    list.print();

    list.insert_after(20, 25);
    cout << "insert_after(20, 25)：";
    list.print();

    list.remove(10);
    cout << "remove(10)：";
    list.print();

    cout << "search(25)："
         << (list.search(25) ? "找到" : "找不到") << "\n";

    list.reverse();
    cout << "reverse()：";
    list.print();

    cout << "\n===== Doubly Linked List 測試 =====\n";
    DoublyLinkedList dlist;
    dlist.push_back(100);
    dlist.push_back(200);
    dlist.push_back(300);
    cout << "正向：";
    dlist.print_forward();
    cout << "反向：";
    dlist.print_backward();

    dlist.insert_after(200, 250);
    cout << "insert_after(200, 250)：";
    dlist.print_forward();

    dlist.remove(200);
    cout << "remove(200)：";
    dlist.print_forward();

    cout << "\n===== Linked Stack 測試 =====\n";
    LinkedStack st;
    st.push(1);
    st.push(2);
    st.push(3);
    cout << "top = " << st.top() << "\n";
    st.pop();
    cout << "pop 後 top = " << st.top() << "\n";

    cout << "\n===== Linked Queue 測試 =====\n";
    LinkedQueue q;
    q.push(11);
    q.push(22);
    q.push(33);
    cout << "front = " << q.front() << "\n";
    q.pop();
    cout << "pop 後 front = " << q.front() << "\n";

    return 0;
}

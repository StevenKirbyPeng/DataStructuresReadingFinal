完整可執行 C++ 程式：Trees 觀念整合

此程式整合本檔常見的樹操作：preorder、inorder、postorder、level-order、BST insert/search、heap push/pop。可直接複製到線上編輯器執行。

Chapter5_Trees_Demo.cpp
#include <iostream>
#include <queue>
#include <vector>
#include <algorithm>
using namespace std;

// 二元樹節點：包含資料、左子樹指標、右子樹指標
struct Node {
    int data;
    Node* left;
    Node* right;
    Node(int x) : data(x), left(nullptr), right(nullptr) {}
};

// Preorder：VLR，先拜訪自己，再左，再右
void preorder(Node* root) {
    if (!root) return;
    cout << root->data << ' ';
    preorder(root->left);
    preorder(root->right);
}

// Inorder：LVR，BST 的 inorder 會得到由小到大順序
void inorder(Node* root) {
    if (!root) return;
    inorder(root->left);
    cout << root->data << ' ';
    inorder(root->right);
}

// Postorder：LRV，常用於釋放整棵樹
void postorder(Node* root) {
    if (!root) return;
    postorder(root->left);
    postorder(root->right);
    cout << root->data << ' ';
}

// Level-order：使用 queue，一層一層拜訪
void levelorder(Node* root) {
    if (!root) return;
    queue<Node*> q;
    q.push(root);
    while (!q.empty()) {
        Node* cur = q.front(); q.pop();
        cout << cur->data << ' ';
        if (cur->left) q.push(cur->left);
        if (cur->right) q.push(cur->right);
    }
}

// BST 插入：小的往左，大的往右
Node* bstInsert(Node* root, int x) {
    if (!root) return new Node(x);
    if (x < root->data) root->left = bstInsert(root->left, x);
    else if (x > root->data) root->right = bstInsert(root->right, x);
    return root;
}

// BST 搜尋：每次比較後只走一邊，因此時間 O(h)
bool bstSearch(Node* root, int x) {
    while (root) {
        if (x == root->data) return true;
        if (x < root->data) root = root->left;
        else root = root->right;
    }
    return false;
}

// Max heap push：放到最後，再 bubble up
void heapPush(vector<int>& heap, int x) {
    heap.push_back(x);
    int i = (int)heap.size() - 1;
    while (i > 1 && heap[i / 2] < heap[i]) {
        swap(heap[i / 2], heap[i]);
        i /= 2;
    }
}

// Max heap pop：刪 root，用最後元素補 root，再 trickle down
void heapPop(vector<int>& heap) {
    if (heap.size() <= 1) return;
    heap[1] = heap.back();
    heap.pop_back();
    int i = 1;
    while (true) {
        int left = i * 2, right = i * 2 + 1, largest = i;
        if (left < (int)heap.size() && heap[left] > heap[largest]) largest = left;
        if (right < (int)heap.size() && heap[right] > heap[largest]) largest = right;
        if (largest == i) break;
        swap(heap[i], heap[largest]);
        i = largest;
    }
}

int main() {
    // 建立一棵 BST
    vector<int> data = {8, 4, 12, 2, 6, 10, 14};
    Node* root = nullptr;
    for (int x : data) root = bstInsert(root, x);

    cout << "Preorder: "; preorder(root); cout << '\n';
    cout << "Inorder: "; inorder(root); cout << '\n';
    cout << "Postorder: "; postorder(root); cout << '\n';
    cout << "Level-order: "; levelorder(root); cout << '\n';
    cout << "Search 10: " << (bstSearch(root, 10) ? "found" : "not found") << '\n';

    // Max heap 示範；heap[0] 不使用
    vector<int> heap = {0};
    for (int x : {15, 10, 20, 8, 12, 25}) heapPush(heap, x);
    cout << "Heap after push: ";
    for (int i = 1; i < (int)heap.size(); ++i) cout << heap[i] << ' ';
    cout << '\n';
    heapPop(heap);
    cout << "Heap after pop max: ";
    for (int i = 1; i < (int)heap.size(); ++i) cout << heap[i] << ' ';
    cout << '\n';
    return 0;
}

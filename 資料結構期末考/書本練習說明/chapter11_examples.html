// 從樹根往下尋找資料加入節點，將資料新增於B-tree中，參數p為目前所在節點，
// xr記錄資料所對應的子鏈結
int Btree::topdown(int new_id, char *new_name, int new_score, Node_type *p,
                      int *x_id, char *x_name, int *x_score, Node_type **xr)
{
    int k;          // 宣告變數，用來保存演算法執行過程中的資料。
    if (p == NULL) { // p為NULL表示新增第一筆資料  // 條件判斷，依照目前資料狀態選擇不同處理路徑。
        *x_id = new_id;
        strcpy(x_name, new_name);
        *x_score = new_score;
        *xr = NULL;
        return 1;  // 回傳函式結果，結束目前函式。
    }
    else {  // 否則分支，處理上一個條件不成立的情況。
        if (search_node(new_id, p, &k)) {   // 條件判斷，依照目前資料狀態選擇不同處理路徑。
		    //找尋新增資料鍵值是否重覆，若重覆則顯示錯誤
            cout << "  ID number has existed!!\n\n";  // 輸出結果，將資料顯示或寫入檔案。
            quit();
            return 0;  // 回傳函式結果，結束目前函式。
        }
        // 繼續往下找尋新增節點
        if(topdown(new_id, new_name, new_score, p->link[k],  // 條件判斷，依照目前資料狀態選擇不同處理路徑。
                                x_id, x_name, x_score, xr))
        {
                // 若新增節點有足夠的空間存放資料，則將資料直接加入該節點
            if (p->count < MAX) {  // 條件判斷，依照目前資料狀態選擇不同處理路徑。
                putdata(*x_id, x_name, *x_score, *xr, p, k);
                return 0;  // 回傳函式結果，結束目前函式。
            }
            else { // 若無足夠空間，則須劃分節點  // 否則分支，處理上一個條件不成立的情況。
                broken(*x_id, x_name, *x_score, *xr, p, k, x_id,
                                            x_name, x_score, xr);
                return 1;  // 回傳函式結果，結束目前函式。
            }
        }
        else  // 否則分支，處理上一個條件不成立的情況。
            return 0;  // 回傳函式結果，結束目前函式。
    }
}

// 將新增資料直接加入於節點中，xr為新增資料對應的子鏈結所在，p為資料加入的節點
void Btree::putdata(int x_id, char *x_name, int x_score, Node_type *xr,
                                Node_type *p, int k)
{
    int i;  // 宣告變數，用來保存演算法執行過程中的資料。
        
    // 將節點中的資料逐一右移，以空出新增資料加入的位置
    for (i = p->count; i > k; i--) {  // for 迴圈，用固定範圍重複走訪資料。
        p->id[i+1] = p->id[i];  // 指定或更新變數值，改變目前資料狀態。
        strcpy(p->name[i+1], p->name[i]);
        p->score[i+1] = p->score[i];  // 指定或更新變數值，改變目前資料狀態。
        p->link[i+1] = p->link[i];  // 指定或更新變數值，改變目前資料狀態。
    }

            getright(p, k+1);

// 將三個節點中的資料合併至一個節點中
void Btree::combine(Node_type *p, int k)  // 函式宣告或函式開始，負責執行一段可重複使用的演算法步驟。
{
    int c;  // 宣告變數，用來保存演算法執行過程中的資料。
    Node_type *l, *q;
        
    q = p->link[k];  // 指定或更新變數值，改變目前資料狀態。
    l = p->link[k-1];  // 指定或更新變數值，改變目前資料狀態。
    l->count++;
    l->id[l->count] = p->id[k];  // 指定或更新變數值，改變目前資料狀態。
    strcpy(l->name[l->count], p->name[k]);
    l->score[l->count] = p->score[k];  // 指定或更新變數值，改變目前資料狀態。
    l->link[l->count] = q->link[0];  // 指定或更新變數值，改變目前資料狀態。
    for (c = 1; c <= q->count; c++) {  // for 迴圈，用固定範圍重複走訪資料。
        l->count++;
        l->id[l->count] = q->id[c];  // 指定或更新變數值，改變目前資料狀態。
        strcpy(l->name[l->count], q->name[c]);
        l->score[l->count] = q->score[c];  // 指定或更新變數值，改變目前資料狀態。
        l->link[l->count] = q->link[c];  // 指定或更新變數值，改變目前資料狀態。
    }
    for (c = k; c < p->count; c++) {  // for 迴圈，用固定範圍重複走訪資料。
        p->id[c] = p->id[c+1];  // 指定或更新變數值，改變目前資料狀態。
        strcpy(p->name[c], p->name[c+1]);
        p->score[c] = p->score[c+1];  // 指定或更新變數值，改變目前資料狀態。
        p->link[c] = p->link[c+1];  // 指定或更新變數值，改變目前資料狀態。
    }

}

  8. 變數命名通常反映它在演算法中的任務。例如 size、n、capacity 多半代表資料數量或容量；top、front、rear、current 多半代表目前位置；temp、p、q、prev 多半代表暫存或指標移動。
  9. 核心迴圈是本程式最重要的地方。它通常會反覆檢查條件、更新索引、比較資料或搬移資料。理解迴圈時，要先問三件事：迴圈從哪裡開始、什麼時候停止、每次執行後哪些變數會改變。
  10. 判斷式 if / else 是演算法決策點。它通常負責處理特殊情況，例如資料為空、找到目標、索引越界、需要交換、需要遞迴、需要往左或往右走。考試常會要求你說明某個 if 條件為什麼必要。
  11. 如果程式使用遞迴，必須先找終止條件。終止條件通常對應最簡單情況，例如資料為空、左右邊界交錯、節點為 NULL 或只剩一個元素。沒有正確終止條件就會造成無窮遞迴。
  12. 如果程式使用排序或搜尋，必須注意資料是否需要事先有序。像二元搜尋必須建立在排序後的資料上；若資料未排序，即使程式語法正確，結果仍可能錯誤。這也是資料結構題目常見陷阱。
  13. 如果程式使用堆疊，重點是 LIFO：最後放入的資料最先取出。這種特性常用於括號配對、函式呼叫、DFS、回溯與運算式轉換。trace 時可以畫出 top 從哪裡移動到哪裡。
  14. 如果程式使用佇列，重點是 FIFO：最早放入的資料最早取出。這種特性常用於 BFS、排程、等待線與層序走訪。trace 時要同時記錄 front 與 rear，避免混淆佇列頭尾。
  15. 如果程式使用樹，必須注意每個節點與子節點的關係。二元搜尋樹依照 key 大小決定往左或往右；heap 則依照父子大小維持優先權；一般二元樹走訪則依照 LVR、VLR、LRV 決定輸出順序。
  16. 如果程式使用圖，重點是 visited 陣列與相鄰節點。visited 可以避免重複拜訪或無限循環；相鄰表示法會影響時間複雜度。相鄰矩陣適合密集圖，相鄰串列適合稀疏圖。
  17. 本程式的時間複雜度可以從「主要迴圈執行幾次」估計。如果只有單層掃描通常是 O(n)；雙層迴圈常是 O(n²)；每次把資料範圍折半常是 O(log n)；樹高控制得好時常接近 O(log n)。
  18. 空間複雜度要看是否建立額外資料結構。若只使用少數變數，通常是 O(1)；若建立與輸入大小相同的陣列、堆疊、佇列或 visited，通常是 O(n)；二維矩陣常是 O(mn)。
  19. 記憶體配置是本程式的重要觀察點。靜態陣列大小固定但簡單；動態陣列可依需求配置但要注意 delete；鏈結串列彈性高，但每個節點都需要額外指標空間。
  20. 程式執行結果要和演算法目的互相對照。輸出不是只看有沒有跑出東西，而是要確認每一筆結果是否符合資料結構規則。例如排序結果是否遞增、搜尋位置是否正確、樹走訪順序是否合理。
  21. 常見錯誤包含：陣列索引越界、指標沒有初始化、刪除節點後仍使用原指標、遞迴沒有終止條件、忘記更新 front/rear/top、比較條件寫反、或是沒有處理空資料結構。
  22. 老師出 trace code 題目時，通常會要求你寫出每一步變數變化。建議用表格記錄迴圈次數、目前索引、比較對象、資料結構內容與輸出結果，這樣最不容易漏掉中間狀態。
  23. 面試或考試常問這個程式的優缺點。優點通常是觀念清楚、容易實作、適合作為基本資料結構練習；缺點則可能是效率不一定最佳、記憶體使用較多、或遇到極端輸入時需要額外處理。
  24. 實際應用上，這類程式可延伸到資料整理、搜尋系統、排程管理、模擬問題、路徑探索、資料索引與演算法效能比較。資料結構的重點不是背程式，而是知道什麼情境該使用什麼結構。

/* 將資料加入，並調整為M-WAY TREE，NUM為新增之資料鍵值 */

{

        ptr->n++;
        root = ptr;  // 指定或更新變數值，改變目前資料狀態。
    }
    else {  // 否則分支，處理上一個條件不成立的情況。

            cout << " Number %d has existed!!\n"<< num;  // 輸出結果，將資料顯示或寫入檔案。

                for (i = 1; i < MAX-1; i++)  // for 迴圈，用固定範圍重複走訪資料。
                    if (num < node->key[i])  // 條件判斷，依照目前資料狀態選擇不同處理路徑。
                        break;
                moveright(i, num);

                initial();

                    if(num < prev->key[i])  // 條件判斷，依照目前資料狀態選擇不同處理路徑。

            }
        }
    }
}

/* 將資料移除，並調整為M-WAY TREE，NODETEMP為刪除資料所在節點，
 LOCATION為資料在節點中的位置 */
void Mwaytree::removes(NodeType *nodeTemp, int location)  // 函式宣告或函式開始，負責執行一段可重複使用的演算法步驟。
{
    int i;  // 宣告變數，用來保存演算法執行過程中的資料。
    node = nodeTemp;  // 指定或更新變數值，改變目前資料狀態。
    replace = findNext(node->link[location]);/*找尋替代之後繼節點*/  // 指定或更新變數值，改變目前資料狀態。
    if (replace == NULL) {  /* 沒有後繼節點的狀況 */  // 條件判斷，依照目前資料狀態選擇不同處理路徑。
        replace = findPrev(node->link[location-1]);  // 指定或更新變數值，改變目前資料狀態。
        /* 找尋替代之前行節點 */
        if (replace == NULL) {  /* 沒有前行節點的狀況 */  // 條件判斷，依照目前資料狀態選擇不同處理路徑。
            moveleft(location);
            replace = node;  // 指定或更新變數值，改變目前資料狀態。
            if (node->n == 0) {  /* 刪除資料後，節點內資料為0的處理 */  // 條件判斷，依照目前資料狀態選擇不同處理路徑。
                if (node == root)  /* 當節點為根的狀況 */  // 條件判斷，依照目前資料狀態選擇不同處理路徑。
                    root = NULL;  // 指定或更新變數值，改變目前資料狀態。
                else  /* 節點不是根，則調整鏈結 */  // 否則分支，處理上一個條件不成立的情況。
                    for (i = 0; i <= parent->n; i++)  // for 迴圈，用固定範圍重複走訪資料。
                        if (node == parent->link[i]) {  // 條件判斷，依照目前資料狀態選擇不同處理路徑。
                            parent->link[i] = NULL;  // 指定或更新變數值，改變目前資料狀態。
                            break;
                        }
                delete replace;  // 釋放動態記憶體，避免記憶體洩漏。
            }
        }
        else {  /* 有前行節點的狀況 */  // 否則分支，處理上一個條件不成立的情況。
            /* 以前行節點的資料代替刪除資料 */
            node->key[location] = replace->key[replace->n];  // 指定或更新變數值，改變目前資料狀態。
            parent = prev;  // 指定或更新變數值，改變目前資料狀態。
            removes(replace, replace->n);  /* 移除替代資料 */
        }
    }
    else {  /* 有後繼節點的狀況 */  // 否則分支，處理上一個條件不成立的情況。
        /* 以後繼節點的資料代替刪除資料 */
        node->key[location] = replace->key[1];  // 指定或更新變數值，改變目前資料狀態。
        parent = prev;  // 指定或更新變數值，改變目前資料狀態。
        removes(replace, 1);  /* 移除替代資料 */
    }

/* 將節點內資料右移--將節點資料右移至INDEX位置，並將NUM插入 */

{
    int i;  // 宣告變數，用來保存演算法執行過程中的資料。

        node->link[i] = node->link[i-1];  // 指定或更新變數值，改變目前資料狀態。
    }

    /* 調整NUM左右鏈結 */
